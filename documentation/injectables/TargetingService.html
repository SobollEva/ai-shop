import { HttpClient } from '@angular/common/http';
import { Inject, Injectable } from '@angular/core';
import { Observable } from 'rxjs/Observable';
import { AutorizationShopService } from '../autorization/autorization-shpf/autorization-shpf.service';
import { SHOPIFY_BASE_URL } from '../shared/shared';
import { AdAccountInfo, CustomerInfo, UploadVideoStatus } from '../shared/shared.type';
import { VideoRenderService } from '../video-render/video-render.service';
import * as moment from 'moment';
import * as param from 'jquery-param';

@Injectable()
export class TargetingService {

  constructor(@Inject(SHOPIFY_BASE_URL) private host,
              private http: HttpClient,
              private videoRenderService: VideoRenderService,
              private autorizationShpfService: AutorizationShopService) {
  }

// TODO: разобраться с пустыми значениями
  public adAccountSelected: AdAccountInfo;
  public fbPageSelected: CustomerInfo = {id: '', name: ''};
  public campaignName = 'Video_Ad_1';
  public targetingGeo = {
    geo_locations: this.autorizationShpfService.shopifyInfo.country
      ? {
        location_types: ['home', 'recent'],
        countries: [this.autorizationShpfService.shopifyInfo.country]
      }
      : {location_types: ['home', 'recent']},
    excluded_geo_locations: {}
  };
  public dailyBudget = 20;
  public dateStop: moment.Moment = moment()
    .add(7, 'days');
  public dateStart: moment.Moment = moment();
  public gender = 'all';
  public minAge = 18;
  public maxAge = 65;
  public video = {
    video_id: '',
    image_url: ''
  };

  public targetingDetailed = {
    flexible_spec: [{}],
    exclusions: {}
  };

  public getFbPageList$(): Observable<CustomerInfo[]> {
    return this.http.get<CustomerInfo[]>(`${this.host}/api/shopify/pages`);
  }

  public getAdAccountList$(): Observable<{ data: AdAccountInfo[] }> {
    return this.http.get<{ data: AdAccountInfo[] }>(`${this.host}/api/shopify/adaccounts`);
  }

  public uploadVideoToFb(): Observable<{ id: string }> {
    return this.http.post<{ id: string }>(`${this.host}/api/shopify/video/upload`,
      param({adaccount_id: this.adAccountSelected.id, file_url: this.videoRenderService.getVideo}));
  }

  public getUploadStatus(videoId: string): Observable<UploadVideoStatus> {
    return this.http.get<UploadVideoStatus>(`${this.host}/api/shopify/video/upload/${videoId}`);
  }

  public createCampaign(campaignData: any): Observable<any> {
    return this.http.post<any>(`${this.host}/api/shopify/bulk/create`, param(campaignData));
  }

  // TODO: нужен ли тип данных для таргетинга
  public campaignData(): any {
    return {
      id: this.adAccountSelected.id,
      campaigns: [{
        name: `Campaign-${this.campaignName}`,
        objective: 'LINK_CLICKS',
        status: 'ACTIVE',
        adsets: [{
          name: `Adset-${this.campaignName}`,
          status: 'ACTIVE',
          optimization_goal: 'LINK_CLICKS',
          billing_event: 'IMPRESSIONS',
          daily_budget: Number(this.dailyBudget),
          bid_strategy: 'LOWEST_COST_WITHOUT_CAP',
          start_time: this.dateStart.format('YYYY-MM-DDT23:59:59+03:00'),
          end_time: this.dateStop.format('YYYY-MM-DDT23:59:59+03:00'),
          ads: [{
            name: `Ad-${this.campaignName}`,
            status: 'ACTIVE',
            creative: {
              object_story_spec: {
                page_id: this.fbPageSelected.id,
                video_data: {
                  call_to_action: {
                    value: {
                      link: `http://${this.autorizationShpfService.shopifyInfo.domain}`
                    },
                    type: 'SHOP_NOW'
                  },
                  image_url: this.video.image_url,
                  video_id: this.video.video_id
                }
              }
            }
          }],
          targeting: {
            age_max: Number(this.maxAge),
            age_min: Number(this.minAge),
            publisher_platforms: ['facebook'],
            facebook_positions: ['feed'],
            device_platforms: ['mobile', 'desktop'],
            genders: this.gender === 'all' ? [] : [this.gender],
            geo_locations: this.targetingGeo.geo_locations,
            excluded_geo_locations: this.targetingGeo.excluded_geo_locations,
            flexible_spec: Boolean(this.targetingDetailed.flexible_spec) ? this.targetingDetailed.flexible_spec : [{}],
            exclusions: Boolean(this.targetingDetailed.exclusions) ? this.targetingDetailed.exclusions : {}
          }
        }]
      }]
    };
  }
}
