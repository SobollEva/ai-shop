import { HttpClient } from '@angular/common/http';
import { Inject, Injectable } from '@angular/core';
import { BehaviorSubject, Observable } from 'rxjs';
import { skip } from 'rxjs/operators';
import { MetricsService } from '../core/metrics.service';
import { SHOPIFY_BASE_URL } from '../shared/shared';
import { AuthorizationFbService } from './autorization-fb.service';
import { AutorizationShopService } from './autorization-shpf/autorization-shpf.service';
import { AuthorizationUser } from './autorization.type';

@Injectable()
export class ShopifyAuthorizationService {
  private isAutorization$$: BehaviorSubject<boolean> = new BehaviorSubject<boolean>(null);
  public isAutorization$: Observable<boolean> = this.isAutorization$$.asObservable()
    .pipe(skip(1));
// TODO: нужна эта переменная?
  public isAutorization = false;

  constructor(@Inject(SHOPIFY_BASE_URL) private host,
              private http: HttpClient,
              private autorizationShpfService: AutorizationShopService,
              private autorizationFbService: AuthorizationFbService,
              private metricsService: MetricsService) {
    this.getUser();
  }

  /**
   * Method for checking autorization
   */
  public getUser(): void {
    this.http.get<AuthorizationUser>(`${this.host}/api/shopify/info`)
      .subscribe((response: AuthorizationUser) => {
        if (!response) {
          this.autorizationShpfService.getAutorizationShpfLogin();
          this.isAutorization = false;
          this.isAutorization$$.next(false);
        } else if (!Object.keys(response)
          .includes('user')) {
          this.autorizationFbService.getAutorizationFbLogin();
          this.isAutorization = false;
          this.isAutorization$$.next(false);
        } else if (!Object.keys(response)
          .includes('shop')) {
          this.autorizationShpfService.getAutorizationShpfLogin();
          this.isAutorization = false;
          this.isAutorization$$.next(false);
        } else {
          this.autorizationShpfService.shopifyInfo = response.shop;
          this.isAutorization = true;
          this.isAutorization$$.next(true);
          this.metricsService.isImpersonated = response.user.impersonated;
        }
      });
  }
}
