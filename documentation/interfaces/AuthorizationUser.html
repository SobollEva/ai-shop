export interface ShopifyUser {
  name: string;
  title: string;
  domain: string;
  country: string;
  currency: string;
}

export interface FacebookUser {
  id: number;
  username: string;
  email: string;
  roles: RoleName[];
  facebook_uid: string;
  facebook_name: string;
  impersonated: boolean;
  previous_user: null;
}

export interface AuthorizationUser {
  shop?: ShopifyUser;
  user?: FacebookUser;
  dashboard?: null;
  status?: null;
}

export interface ShopifyParams {

  hmac?: string;
  shop: string;
  state?: number;
  timestamp: number;
}
