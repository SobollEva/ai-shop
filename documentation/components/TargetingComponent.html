import { Component, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { MetricsService } from '../core/metrics.service';
import { AdAccountInfo, CustomerInfo } from '../shared/shared.type';
import { CampaignDialogComponent } from './campaign-dialog/campaign-dialog.component';
import { TargetingService } from './targeting.service';
import { MAT_DATE_FORMATS, MatDatepicker } from '@angular/material';
import { getCurrencySymbol } from '@angular/common';
import * as moment from 'moment';

export const DATE_FORMATS = {
  parse: {
    dateInput: 'll',
  },
  display: {
    dateInput: 'll',
    monthYearLabel: 'MMM YYYY',
    dateA11yLabel: 'll',
    monthYearA11yLabel: 'MMMM YYYY',
  },
};

@Component({
  selector: 'app-targeting',
  templateUrl: './targeting.component.html',
  styleUrls: ['./targeting.component.css'],
  providers: [
    {
      provide: MAT_DATE_FORMATS,
      useValue: DATE_FORMATS
    },
  ]
})
export class TargetingComponent implements OnInit {
  public fbPageList: CustomerInfo[];
  public adAccountList: AdAccountInfo[];
  public ageList: number[] = [];
  public currencySymbol: any;
  public isRightBudget = true;

  constructor(public targetingService: TargetingService,
              public campaignDialog: MatDialog,
              public metricsService: MetricsService) { }

  ngOnInit() {
    // TODO:          
    this.targetingService.getFbPageList$()
      .subscribe((fbPageList: CustomerInfo[]) => {
        this.fbPageList = fbPageList;
        this.targetingService.fbPageSelected = fbPageList[0];
      });

    this.targetingService.getAdAccountList$()
      .subscribe((adAccountList: { data: AdAccountInfo[] }) => {
        this.adAccountList = adAccountList.data;
        this.targetingService.adAccountSelected = this.adAccountList[0];
        this.currencySymbol = getCurrencySymbol(this.targetingService.adAccountSelected.currency, 'narrow');
      });

    for (let i: number = this.targetingService.minAge; i <= this.targetingService.maxAge; i++) {
      this.ageList.push(i);
    }
  }

  public openCalendar(event: MouseEvent, datepicker: MatDatepicker<moment.Moment>): void {
    event.preventDefault();
    datepicker.open();
  }

  public openCreateCampaign(): void {
    this.campaignDialog.open(CampaignDialogComponent, {disableClose: true});
  }

  public changeCurrency(adAccountSelect: AdAccountInfo): void {
    this.currencySymbol = getCurrencySymbol(adAccountSelect.currency, 'narrow');
  }

  public checkBudget(): void {
    this.isRightBudget = Boolean(Number(this.targetingService.dailyBudget > 0));
  }

  // TODO:                 
}
