import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { ProductListService } from '../product-list/product-list.service';
import { Product, VideoRender, VideoRenderStatus } from '../shared/shared.type';
import { VideoRenderService } from './video-render.service';
import { interval } from 'rxjs/internal/observable/interval';
import { startWith, switchMap } from 'rxjs/operators';
import { ISubscription } from 'rxjs/Subscription';

@Component({
  selector: 'app-video-render',
  templateUrl: './video-render.component.html',
  styleUrls: ['./video-render.component.css']
})
export class VideoRenderComponent implements OnInit {
  private timer: ISubscription;
  public progressVideo: string = '0';
  // TODO: проверять статусы и выводить ошибки
  constructor(private videoRenderService: VideoRenderService,
              private productListService: ProductListService,
              private router: Router) { }

  ngOnInit() {
    const videoIdList: number[] = this.productListService.productTopList
      .map((product: Product) => product.id);
    this.videoRenderService.createVideoJob(videoIdList)
      .subscribe((response: VideoRender) => {
        this.timer = interval(5000)
          .pipe(
            startWith(0),
            switchMap(() => this.videoRenderService.getVideoStatus(response.id)))
          .subscribe((progress: VideoRenderStatus) => {
            this.progressVideo = progress.progress;
            if (progress.progress === '100') {
              this.timer.unsubscribe();
              if (progress.result) {
                this.videoRenderService.videoSrc$$.next(progress.result);
                this.router.navigate(['review']);
              }
            }
          });
      });
  }
}
