import { Component, HostListener, Input, OnInit } from '@angular/core';
import { Angular2FontawesomeModule } from 'angular2-fontawesome';
import { DropEvent } from 'ng-drag-drop';
import { VideoRenderService } from '../video-render/video-render.service';
import { MetricsService } from '../core/metrics.service';
import { ProductDialogComponent } from './product-dialog/product-dialog.component';
import { ProductListService } from './product-list.service';
import { MatDialog } from '@angular/material/dialog';

@Component({
  selector: 'app-product-list',
  templateUrl: './product-list.component.html',
  styleUrls: ['./product-list.component.css']
})
export class ProductListComponent implements OnInit {
  constructor(public productListService: ProductListService,
              public productDialog: MatDialog,
              public videoRenderService: VideoRenderService,
              public metricsService: MetricsService) { }

  public removeProductPosition: number;
// TODO: Angular2FontawesomeModule
// TODO: посмотреть все методы drag - может есть событие единожды
  @HostListener('drag', ['event'])
  onDrag() {
    this.removeProductPosition = +event.srcElement.className.split('-')[1];
  }

  ngOnInit() {
    this.productListService.getProductTopList();
  }

  public openProductDialog(index: number): void {
    const dialogRef = this.productDialog.open(ProductDialogComponent, {data: {index: index}});
  }

  public deleteProductFromTopList(index: number): void {
    this.productListService.deleteProductFromTopList(index);
  }

  public onProductDrop(event: DropEvent): void {
    //document.querySelector('.product-item_content:hover').classList
    const product = event.dragData;
    const pastPosition = +event.nativeEvent.srcElement.className.split('-')[1];
    if (pastPosition !== this.removeProductPosition) {
      if (this.removeProductPosition > pastPosition) {
        this.productListService.productTopList.splice(pastPosition, 0, product);
        this.productListService.productTopList.splice(this.removeProductPosition + 1, 1);
      } else {
        this.productListService.productTopList.splice(pastPosition + 1, 0, product);
        this.productListService.productTopList.splice(this.removeProductPosition, 1);
      }
    }
  }
}
