import { Component, OnInit, ViewChild } from '@angular/core';
import { Inject } from '@angular/core';
import { MatDialogRef } from '@angular/material/dialog';
import { MatPaginator } from '@angular/material';
import { MetricsService } from '../../core/metrics.service';
import { Product } from '../../shared/shared.type';
import { ProductListComponent } from '../product-list.component';
import { ProductListService } from '../product-list.service';
import { MAT_DIALOG_DATA } from '@angular/material';
import { merge, of as observableOf } from 'rxjs';
import { catchError, map, startWith, switchMap } from 'rxjs/operators';

@Component({
  selector: 'app-product-dialog',
  templateUrl: './product-dialog.component.html',
  styleUrls: ['./product-dialog.component.css']
})
export class ProductDialogComponent implements OnInit {
  public productList: Product[] = [];
  public isLoadingResults = true;
  public displayedColumns = ['image', 'title', 'price', 'button'];
  @ViewChild(MatPaginator) paginator: MatPaginator;

  constructor(public productListService: ProductListService,
              private dialogRef: MatDialogRef<ProductListComponent>,
              @Inject(MAT_DIALOG_DATA) public data,
              private metricsService: MetricsService) { }

  ngOnInit() {
    if (!this.productListService.productLength) {
      this.productListService.getProductCount();
    }
    merge(this.paginator.page)
      .pipe(
        startWith({}),
        switchMap(() => {
          this.isLoadingResults = true;
          return this.productListService.getProductList(this.paginator.pageIndex + 1);
        }),
        catchError(() => {
          this.isLoadingResults = false;
          return observableOf([]);
        })
      )
      .subscribe((response: Product[]) => {
        this.isLoadingResults = false;
        this.productList = response;
      });
  }

  public addProductToAd(product: Product): void {
    this.metricsService.trackEvent('Products_Button_Add new products');
    this.productListService.productTopList.splice(this.data.index, 1, product);
    this.productListService.buttonDisabled = this.productListService.productTopList
      .some((item: Product) => !item.id);
    this.dialogRef.close();
  }
